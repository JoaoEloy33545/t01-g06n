\subsection*{3.2 Estratégia e Implementação}
\addcontentsline{toc}{subsection}{3.2 Estratégia e Implementação}

\textit{(Descrever as principais decisões de design: escolha de gRPC como protocolo de comunicação, Java/Quarkus como framework, Podman para containerização. Justificar cada escolha no contexto do modelo X/Open.)}

\subsubsection*{Service.Vector (Resource Manager)}

\textit{(Descrever a implementação: staging area por transação — os writes são temporários até \texttt{xa\_commit}; \texttt{xa\_rollback} descarta o staging. Operação \texttt{getSum} para uso pelo Validator. Mostrar excerto do \texttt{Dvector.proto}.)}

\begin{lstlisting}[language=proto, caption={Contrato gRPC do Service.Vector (excerto).}]
service DvectorService {
  rpc Read      (ReadRequest)  returns (ReadResponse);
  rpc Write     (WriteRequest) returns (WriteResponse);
  rpc XaPrepare (XaRequest)   returns (XaResponse);
  rpc XaCommit  (XaRequest)   returns (XaResponse);
  rpc XaRollback(XaRequest)   returns (XaResponse);
  rpc GetSum    (SumRequest)  returns (SumResponse);
}
\end{lstlisting}

\subsubsection*{Service.TPLM (Two-Phase Lock Manager)}

\textit{(Descrever: tabela de locks em memória, granularidade por vetor, compatibilidade read-read vs. read-write/write-write, fila de espera para locks em conflito, deteção de deadlock por timeout. Mostrar excerto do \texttt{tplm.proto}.)}

\begin{lstlisting}[language=proto, caption={Contrato gRPC do Service.TPLM (excerto).}]
service TplmService {
  rpc AcquireLock  (LockRequest) returns (LockResponse);
  rpc ReleaseLocks (TxRequest)   returns (Empty);
  rpc GetLockTable (Empty)       returns (LockTableResponse);
}
\end{lstlisting}

\subsubsection*{Service.TM (Transaction Manager)}

\textit{(Descrever: geração de txId único, registo de RMs participantes, implementação do 2PC — Fase Prepare e Fase Commit/Abort, log persistente de transações para recuperação após falha, coordenação com TPLM para libertar locks após commit/rollback. Mostrar excerto do \texttt{Tm.proto}.)}

\begin{lstlisting}[language=proto, caption={Contrato gRPC do Service.TM (excerto).}]
service TmService {
  rpc BeginTx    (BeginRequest)  returns (TxResponse);
  rpc CommitTx   (TxRequest)    returns (TxResponse);
  rpc AbortTx    (TxRequest)    returns (TxResponse);
  rpc GetTxStatus(GetTxRequest) returns (TxResponse);
}
\end{lstlisting}

\subsubsection*{Fluxo de uma Transferência}

\textit{(Apresentar o diagrama de sequência de uma transferência entre dois vetores — figura \texttt{sequence\_diagram.png}.)}

\begin{lstlisting}[language=bash, caption={Sequência de operações numa transferência (pseudocódigo).}]
Client -> TM:         tx_begin()
Client -> TPLM:       acquireWriteLock(txId, vecId_src, idx)
Client -> TPLM:       acquireWriteLock(txId, vecId_dst, idx)
Client -> Vector_src: read(idx)
Client -> Vector_dst: read(idx)
Client -> Vector_src: write(idx, new_value)
Client -> Vector_dst: write(idx, new_value)
Client -> TM:         tx_commit()
  TM -> Vector_src:     xa_prepare()
  TM -> Vector_dst:     xa_prepare()
  TM -> Vector_src:     xa_commit()
  TM -> Vector_dst:     xa_commit()
  TM -> TPLM:           releaseLocks(txId)
\end{lstlisting}
